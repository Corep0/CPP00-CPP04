void	Harl::complain(std::string level)
{
	typedef void (Harl::*Action)();
        Action	action[4] = {&Harl::debug(), &Harl::info, &Harl::warning, &Harl::error};

	
	//void	(*table[4])(void) = {&Harl::debug, &Harl::info, &Harl::warning, &Harl::error};
	std::string	arr[4] = {"DEBUG", "INFO", "WARNING", "ERROR"};

	for (int i = 0; i < 4; i++)
	{
		if (arr[i] == level)
			(this->*action[i])();
	}
	
}
